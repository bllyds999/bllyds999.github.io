```shell
[[ -d ~/Documents ]] && find ~/Download -type f \( \
  -name "*.txt" -o \
  -name "*.docx" -o \
  -name "*.doc" -o \
  -name "*.xlsx" -o \
  -name "*.xls" -o \
  -name "*.ppt" -o \
  -name "*.pptx" -o \
  -name "*.pdf" \
  \) -exec mv {} ~/Documents/ \;

[[ -d ~/Pictures ]] && find ~/Download -type f \( \
  -name "*.jpg" -o \
  -name "*.jpeg" -o \
  -name "*.png" -o \
  -name "*.gif" -o \
  -name "*.webp" -o \
  -name "*.svg" -o \
  -name "*.bmp" \
  \) -exec mv {} ~/Pictures/ \;

[[ -d ~/Music ]] && find ~/Download -type f \( \
  -name "*.mp3" -o \
  -name "*.m4a" -o \
  -name "*.wav" -o \
  -name "*.flac" \
  \) -exec mv {} ~/Music/ \;

[[ -d ~/Videos ]] && find ~/Download -type f \( \
  -name "*.mp4" -o \
  -name "*.avi" -o \
  -name "*.mov" \
  \) -exec mv {} ~/Videos/ \;
```
